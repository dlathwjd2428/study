#include <stdio.h>
#include <Windows.h>
#include <conio.h>

int i = 10;
void Draw(int a[][10])
{
	system("cls");
	for (int y = 0; y < 10; y++)
	{
		for (int x = 0; x < 10; x++)
		{
			if (a[y][x] == 1 || a[y][x] == 2)
				printf("☆");
			else
				printf("  ");
		}
		printf("\n");
	}
	printf("별 멈추기 = d");
}

void Update(int a[][10])
{
	for (int y = 0; y < 10; y++)
	{
		for (int x = 0; x < 10; x++)
		{
			if (a[y][x] == 1)
			{
				if (x - 1 < 0)
				{
					a[y][x] = 0;
					a[y][9] = 1;
				}
				else
				{
					a[y][x] = 0;
					a[y][x - 1] = 1;
				}
			}
		}
	}
}

void Stop(int a[][10])
{
	for (int j = 0; j < 10; j++)
	{
		if (a[i][j] == 1)
		{
			a[i][j] = 2;
			return;
		}
	}
}

void main()
{
	int a[10][10] = { 0 };
	char ch;
	for (int y = 0; y < 10; y++)
	{
		for (int x = 0; x < 10; x++)
		{
			if (x == 9)
				a[y][x] = 1;
		}
	}
	printf("별을 보시겠습니까??\n");
	system("pause");
	system("cls");
	while (1)
	{
		Update(a);
		if (kbhit())
		{
			ch = getch();
			if (ch == 'd' || ch == 'D')
			{
				i--;
				Stop(a);
			}
		}
		Draw(a);
		Sleep(100);
	}
}
