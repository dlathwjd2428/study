//init 함수
void init()
{
	for(int y = 0; y < HEIGHT; y++)
	{
		for(int x = 0; x < WEIGHT; x++)
		{
			if(x == 0 || x == 9) //첫열과 끝열은 벽으로
				a[y][x] = WALL; //무조건고정이라서
		}
	}
	characterX = WEIGHT / 2;
	a[CHARACTERY][characterX] = CHARACTER;
}

}
//move 조작함수
void move()
{
	char ch;
	ch = getch();
	a[CHARACTERY][characterX] = NULL;
	switch(ch)
	{
		case LEFT:
			if(a[CHARACTERY][characterX -1] != WALL)
				characterX--;
			break;
		case RIGHT:
			if(a[CHARACTERY][characterX + 1] != WALL)
				characterX++;
			break;
	}
	movecheck();
	a[CHARACTERY][characterX + 1] = CHARACTER;

}

//draw 함수 
void draw() // m.o
{
	system("cls");
	for(int y = 0; y < HEIGHT; y++)
	{
		for(int x = 0; x < WEIGHT; x++) //각자리값마다 뭐그려넣을지 지정
		{
			if(a[y][x] == WALL) //벽그리기
				printf("▦");
			else if(a[y][x] == STAR) //별그리기
				printf("★");
			else if(a[y][x] == CHARACTER) //캐릭터그리기
				printf("▲");
			else
				printf("  ");
		}
		printf("\n"); //개행
	}
}
