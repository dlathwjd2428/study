#include<stdio.h>

int Strcmp(char a[], char b[])
{
	int i;

	for(i = 0;; i++)
	{
		if(a[i] == b[i] && a[i] == NULL)
			return 0;
		else if(a[i] > b[i])
			return 1;
		else if(a[i] < b[i])
			return -1;
	}

}
void main()
{
	char str1[10] = "string!!";
	char str2[10] = "string";

	printf("%s==%s:%d\n", str1, str2, Strcmp(str1, str2));
	printf("%s==%s:%d\n", "abc", "abc", Strcmp("abc", "abc"));
	printf("%s==%s:%d\n", "abc", "def", Strcmp("abc", "def"));
}
